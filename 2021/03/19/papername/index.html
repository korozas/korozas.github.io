        pre = head;

                pre->next = p->next; //交换节点方法
                temp = p->next->next;
                p->next->next = p;
                p->next = temp;
                p = pre->next;  //p回退一个节点
            &#125;
            p = p->next;  //p再前进一个节点
            pre = pre->next;
